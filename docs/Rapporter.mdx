import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';
import { faDownload } from '@fortawesome/free-solid-svg-icons';
import styles from './css/Rapporter.module.css';

export const DownloadButton = ({ filename, url }) => {
  const handleClick = () => window.open(url, '_blank');
  return (
    <button
      onClick={handleClick}
      className={styles.download}>
      <span>{filename}<FontAwesomeIcon icon={faDownload} style={{ marginRight: '5px' }} /></span>
    </button>
  );
};


<div className={styles.rapporter}>
    <DownloadButton filename={"Halvårsrapport 2024"} url={'/dataset/H1 Rapport Fintech Enigma.pdf'} />
    <DownloadButton filename={"Årsrapport 2023"} url={'/dataset/Årsrapport23.pdf'} />
</div>
